   11 expandDefinitions :: MonadSafe m => InstructionList -> m LambdaList
   12 expandDefinitions il = flip expandDefinitionsByMap il =<< buildDefinitionMap il
   13 
   14 removeDefinitions :: InstructionList -> LambdaList
   15 removeDefinitions l = (maybeToList . removeDefinition) =<< l
   16 
   17 removeDefinition :: Instruction -> Maybe Lambda
   18 removeDefinition (Def _ _) = Nothing
   19 removeDefinition (Eval  l) = Just l
   20 
   21 buildDefinitionMap :: MonadSafe m => InstructionList -> m Definitions
   22 buildDefinitionMap il = nextDefinitionMap il Map.empty
   23 
   24 nextDefinitionMap :: MonadSafe m => InstructionList -> Definitions -> m Definitions
   25 nextDefinitionMap              [] m = pure m
   26 nextDefinitionMap d@(Def n f : l) m = nextDefinitionMap l =<< appendError (show d) (ff <$> expandDefinitionByMap m f) where ff x = insert n x m
   27 nextDefinitionMap   (_       : l) m = nextDefinitionMap l m
   28 
   29 expandDefinitionsByMap :: MonadSafe m => Definitions -> InstructionList -> m LambdaList
   30 expandDefinitionsByMap m il = appendError (show il) $ sequence (expandDefinitionByMap m <$> removeDefinitions il)
   31 
   32 expandDefinitionByMap :: MonadSafe m => Definitions -> Lambda -> m Lambda
   33 expandDefinitionByMap m (App f g) = liftA2 App (expandDefinitionByMap m f) (expandDefinitionByMap m g)
   34 expandDefinitionByMap m (Abs n f) = Abs n <$> expandDefinitionByMap m f
   35 expandDefinitionByMap m (Var n)   = liftMaybeOrErrorTupleList [("n", show n) , ("m" , show m)] $ lookup n m
   36 expandDefinitionByMap _         l = pure l
