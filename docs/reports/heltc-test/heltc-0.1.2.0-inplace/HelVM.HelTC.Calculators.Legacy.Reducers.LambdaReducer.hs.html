    7 reduceLambda :: Lambda -> Lambda

   14 onlySKI :: Lambda -> Bool
   15 onlySKI S         = True
   16 onlySKI K         = True
   17 onlySKI I         = True
   18 onlySKI (App f a) = onlySKI f && onlySKI a
   19 onlySKI _         = False

   21 reduceToSki :: Text -> Lambda -> Lambda
   22 reduceToSki _ e
   23   | onlySKI e             = App K e
   24 reduceToSki x0 (App f a)  = App (App S $ reduceToSki x0 f) $ reduceToSki x0 a
   25 reduceToSki x0 (Abs x1 e) = reduceFun x0 x1 e
   26 reduceToSki x0 (Var x1)   = reduceVar x0 x1
   27 reduceToSki _ _      = error "never occur"
   28 
   29 reduceFun :: Text -> Text -> Lambda -> Lambda
   30 reduceFun x0 x1 e
   31   | x0 == x1  = App K $ reduceToSki x0 e
   32   | otherwise = reduceToSki x0 $ reduceToSki x1 e
   33 
   34 reduceVar :: Text -> Text -> Lambda
   35 reduceVar x0 x1
   36   | x1 == x0  = I
   37   | otherwise = App K (Var x1)
