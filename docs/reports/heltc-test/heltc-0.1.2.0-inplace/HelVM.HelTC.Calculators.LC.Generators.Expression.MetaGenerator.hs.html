    9 generateCodeForIL :: InstructionList -> Text
   10 generateCodeForIL il = unlines $ generateCodeForInstruction <$> il
   11 
   12 generateCodeForInstruction :: Instruction -> Text
   13 generateCodeForInstruction (Def i l) = generateCodeForDefinition i l
   14 generateCodeForInstruction (Eval  l) = generateCodeForLambda l
   15 
   16 generateCodeForDefinition :: Identifier -> Lambda -> Text
   17 generateCodeForDefinition i l = ": " <> i <> " " <> generateCodeForLambda l
   18 
   19 generateCodeForLambda  :: Lambda -> Text
   20 generateCodeForLambda S              = "S"
   21 generateCodeForLambda K              = "K"
   22 generateCodeForLambda I              = "I"
   23 generateCodeForLambda a@(App _ _)    = generateCodeForApp a
   24 generateCodeForLambda f@(Abs _ _)    = generateCodeForAbs f
   25 generateCodeForLambda (AbsRe l f)    = generateCodeForAbsRe l f
   26 generateCodeForLambda (Var v)        = v
   27 generateCodeForLambda (Nat   n)      = show n
   28 generateCodeForLambda (Int (SN s n)) = showSign s <> show n
   29 generateCodeForLambda (Dec s n)      = show s <> show n
   30 generateCodeForLambda (List  l)      = generateCodeForList l
   31 generateCodeForLambda (Str   s)      = generateCodeForSrt s
   32 
   33 generateCodeForApp :: Lambda -> Text
   34 generateCodeForApp (App f a) = generateCodeForApp f <> " " <> par generateCodeForLambda  a
   35 generateCodeForApp e         = par generateCodeForLambda  e
   36 
   37 generateCodeForAbs :: Lambda -> Text
   38 generateCodeForAbs (Abs p e) = "\\ " <> p <> " " <> generateCodeForAbs e
   39 generateCodeForAbs e         = generateCodeForLambda  e
   40 
   41 generateCodeForAbsRe :: IdentifierList -> Lambda -> Text
   42 generateCodeForAbsRe il f = "(lambda (" <> show il <> ")" <> generateCodeForLambda f <> ")" --FIXME Remove AbsRe
   43 
   44 generateCodeForList :: LambdaList-> Text
   45 generateCodeForList l   = "[" <> show l <> "]"
   46 
   47 generateCodeForSrt :: String -> Text
   48 generateCodeForSrt s = "\"" <> toText s <> "\""
