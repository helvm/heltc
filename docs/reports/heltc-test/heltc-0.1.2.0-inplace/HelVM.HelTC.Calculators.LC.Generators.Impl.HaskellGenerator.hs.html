   11 generateCodeForLambda :: Lambda -> Text
   12 generateCodeForLambda S              = "S"
   13 generateCodeForLambda K              = "K"
   14 generateCodeForLambda I              = "I"
   15 generateCodeForLambda a@(App _ _)    = generateCodeForApp a
   16 generateCodeForLambda f@(Abs _ _)    = "\\" <> generateCodeForAbs f
   17 generateCodeForLambda (AbsRe l f)    = generateCodeForAbsRe l f
   18 generateCodeForLambda (Var v)        = v
   19 generateCodeForLambda (Nat   n)      = show n
   20 generateCodeForLambda (Int (SN s n)) = showSign s <> show n
   21 generateCodeForLambda (Dec s n)      = show s <> show n
   22 generateCodeForLambda (List  l)      = generateCodeForList l
   23 generateCodeForLambda (Str   s)      = generateCodeForSrt s
   24 
   25 generateCodeForApp :: Lambda -> Text
   26 generateCodeForApp (App f a) = generateCodeForApp f <> " " <> par generateCodeForLambda a
   27 generateCodeForApp e         = par generateCodeForLambda e
   28 
   29 generateCodeForAbs :: Lambda -> Text
   30 generateCodeForAbs (Abs p e) = p <> " " <> generateCodeForAbs e
   31 generateCodeForAbs e         = "-> " <> generateCodeForLambda e
   32 
   33 generateCodeForAbsRe :: IdentifierList -> Lambda -> Text
   34 generateCodeForAbsRe il f = "(lambda (" <> unwords il <> ") " <> generateCodeForLambda f <> ")" --FIXME Remove AbsRe
   35 
   36 generateCodeForList :: LambdaList-> Text
   37 generateCodeForList l   = "[" <> show l <> "]"
   38 
   39 generateCodeForSrt :: String -> Text
   40 generateCodeForSrt s = "\"" <> toText s <> "\""
