   19 -- | Instruction Parsers
   20 rootLambdaLnParser :: Parser Instruction

   23 rootLambdaParser :: Parser Instruction
   24 rootLambdaParser = parens (defineParser <|> evalParser) <?> "root"
   25 
   26 defineParser :: Parser Instruction
   27 defineParser = liftA2 Def (defineSymbol *> identifierParser) lambdaParser <?> "define"
   28 
   29 evalParser :: Parser Instruction
   30 evalParser = (Eval <$> lambdaParser) <?> "eval"

   32 -- | Lambda Parsers
   33 lambdaParser :: Parser Lambda
   34 lambdaParser = (abstractionParser <|> applicationParser) <?> "lambda"
   35 
   36 abstractionParser :: Parser Lambda
   37 abstractionParser = liftA2 AbsRe paramsParser lambdaParser <?> "abstractionRe"
   38 
   39 paramsParser :: Parser IdentifierList
   40 paramsParser = (lambdaSymbol *> parens (some identifierOrUnderscoreParser)) <?> "params"
   41 
   42 applicationParser :: Parser Lambda
   43 applicationParser = (lambdaFromNonEmpty <$> many1' terminalParser) <?> "application"
   44 
   45 terminalParser :: Parser Lambda
   46 terminalParser = (
   47       parens lambdaParser
   48   <|> listParser
   49   <|> stringParser
   50   <|> charParser
   51   <|> signedParser
   52   <|> naturalParser
   53   <|> variableParser

   56 listParser :: Parser Lambda

   58 
   59 stringParser :: Parser Lambda
   60 stringParser = (Str <$> stringLiteral) <?> "string"
   61 
   62 charParser :: Parser Lambda
   63 charParser = (lambdaFromChar <$> charLiteral') <?> "char"
   64 
   65 signedParser :: Parser Lambda
   66 signedParser = (Int <$> liftA2 SN signParser naturalLiteral) <?> "integer"

   68 naturalParser :: Parser Lambda

   71 variableParser :: Parser Lambda
