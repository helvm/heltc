   16 generateCodeForLambda :: MonadSafe m => GeneratorType -> Lambda -> m Text
   17 generateCodeForLambda MLC      = pure . MetaGenerator.generateCodeForLambda
   18 generateCodeForLambda SLC      = pure . SymbolicGenerator.generateCodeForLambda

   22 -- | UnLambda
   23 generateUnLambda :: MonadSafe m => Lambda -> m Text
   24 generateUnLambda l = appendError (show l) $ generateUnLambdaFix l
   25 
   26 generateUnLambdaFix :: MonadSafe m => Lambda -> m Text
   27 generateUnLambdaFix S         = pure "s"
   28 generateUnLambdaFix K         = pure "k"
   29 generateUnLambdaFix I         = pure "i"
   30 generateUnLambdaFix (App f a) = liftA2 generateUnLambdaApp (generateUnLambdaFix f) (generateUnLambdaFix a)
   31 generateUnLambdaFix other     = liftError $ "generateUnLambda " <> show other
   32 
   33 generateUnLambdaApp :: (Semigroup a, IsString a) => a -> a -> a
   34 generateUnLambdaApp f a = "`" <> f <> a
   35

   40 generateZotFix :: MonadSafe m => Lambda -> m Text
