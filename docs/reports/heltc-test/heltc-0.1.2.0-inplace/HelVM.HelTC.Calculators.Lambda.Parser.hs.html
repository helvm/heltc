   20 parseILFile :: BIO m => ILType -> FilePath -> m InstructionList
   21 parseILFile ilType filePath = parseILText ilType =<< wReadFile filePath

   23 parseILText :: MonadSafe m => ILType -> Text -> m InstructionList
   24 parseILText ilType = fromResult . internalParser ilType . toString
   25 
   26 parseLambdaText :: MonadSafe m => LambdaType -> Text -> m Lambda
   27 parseLambdaText ilType = fromResult . internalParser2 ilType . toString

   29 internalParser :: ILType -> String -> Result InstructionList
   30 internalParser = flip parseString mempty . ilParser
   31 
   32 internalParser2 :: LambdaType -> String -> Result Lambda
   33 internalParser2 = flip parseString mempty . lambdaParser

   35 fromResult :: MonadSafe m => Result a -> m a
   36 fromResult (Success e) = pure e
   37 fromResult (Failure e) = liftError $ show e

   39 ilParser :: ILType -> Parser InstructionList
