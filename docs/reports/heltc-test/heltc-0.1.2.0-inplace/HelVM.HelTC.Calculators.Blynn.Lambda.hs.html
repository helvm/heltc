    6 runLC :: LC -> String -> String
    7 runLC lc = (runCom $ rewrite $ deLam lc)
    8 
    9 data Com = S | K | I | B | C | T
   10   deriving stock (Eq , Ord , Read , Show)
   11 
   12 data CL = Lf Com | CL :# CL | Ext String
   13   deriving stock (Eq , Ord , Read , Show)
   14 
   15 type VarId = String
   16 data LC = Var VarId | Lam VarId LC | LC :@ LC | Other Com
   17   deriving stock (Eq , Ord , Read , Show)
   18 
   19 deLam :: LC -> LC
   20 deLam t = case t of
   21   Lam v u -> case deLam u of
   22     Var w | v == w    -> Other I
   23           | otherwise -> Other K :@ Var w
   24     Other c           -> Other K :@ Other c
   25     x :@ y            -> Other S :@ deLam (Lam v x) :@ deLam (Lam v y)
   26     _                 -> error "error in deLam"
   27   x :@ y  -> deLam x :@ deLam y
   28   _       -> t
   29 
   30 rewrite :: LC -> CL
   31 rewrite t = case deLam t of
   32   Other c -> Lf c
   33   x :@ y  -> rewrite x :# rewrite y
   34   _       -> error "error in rewrite"
   35 
   36 reduce :: CL -> Maybe CL
   37 reduce t = case t of
   38   Lf I :# x           -> Just x
   39   Lf K :# x :# _      -> Just x
   40   Lf T :# x :# y      -> Just (y :# x)
   41   Lf S :# x :# y :# z -> Just (x :# z :# (y :# z))
   42   Lf B :# x :# y :# z -> Just (x      :# (y :# z))
   43   Lf C :# x :# y :# z -> Just (x :# z :#  y      )
   44   _                   -> Nothing
   45 
   46 normalize :: CL -> CL
   47 normalize t0 = down t0 [] up1 where
   48   down t args k = case t of
   49     x :# y -> down x (y:args) k
   50     _      -> k t args
   51 
   52   up1 t args = case reduce t of
   53     Just t' -> down t' args up1
   54     Nothing -> case args of
   55       []   -> down t [] up2
   56       a:as -> up1 (t :# a) as
   57 
   58   up2 t args = case args of
   59     []   -> t
   60     a:as -> up2 (t :# normalize a) as
   61 
   62 run :: CL -> CL
   63 run t0 = down t0 [] where
   64   down t args = case t of
   65     x :# y -> down x (y:args)
   66     _      -> up t args
   67 
   68   up t args = case reduce t of
   69     Just t' -> down t' args
   70     Nothing -> case args of
   71       []   -> t
   72       a:as -> up (t :# a) as
   73 
   74 runM :: Monad m => (CL -> m (Maybe CL)) -> CL -> m CL
   75 runM f t0 = down t0 [] where
   76   down t args = case t of
   77     x :# y -> down x (y:args)
   78     _      -> up t args
   79 
   80   up t args = do
   81     m <- f t
   82     case m of
   83       Just t' -> down t' args
   84       Nothing -> case args of
   85         []   -> pure t
   86         a:as -> up (t :# a) as
   87 
   88 encodeChar :: Char -> CL
   89 encodeChar c = go (fromEnum c) where
   90   go 0 = Lf K
   91   go n = Lf K :# (Lf T :# go (n - 1))
   92 
   93 decodeChar :: CL -> Char
   94 decodeChar t0 = unsafeEnum $ execState (runM red $ t0 :# Ext "Z" :# Ext "S") 0 where
   95   red t = case t of
   96     Ext "S" :# a -> do
   97       modify (1+)
   98       pure $ Just $ a :# Ext "Z" :# Ext "S"
   99     _ -> pure $ reduce t
  100 
  101 encode :: [Char] -> CL
  102 encode ""     = Lf K
  103 encode (c:cs) = Lf K :# (Lf C :# (Lf T :# encodeChar c) :# encode cs)
  104 
  105 decode :: CL -> [Char]
  106 decode t0 = execState (runM red $ t0 :# Ext "nil" :# Ext "cons") id "" where
  107   red t = case t of
  108     Ext "cons" :# x :# xs -> do
  109       modify (. (decodeChar x:))
  110       pure $ Just $ xs :# Ext "nil" :# Ext "cons"
  111     _ -> pure $ reduce t
  112 
  113 runCom :: CL -> String -> String
  114 runCom t s = decode (run (t :# encode s))
