    7 reduceSki :: Lambda -> Lambda
    8 reduceSki = fst . flip reduceSkiWithIndex 0

   10 reduceSkiWithIndex :: Lambda -> Int -> (Lambda , Int)
   11 reduceSkiWithIndex (App f a) n = reduceApp f a n
   12 reduceSkiWithIndex S         n = (reduceS n       , n + 1)
   13 reduceSkiWithIndex K         n = (reduceK n       , n + 1)
   14 reduceSkiWithIndex I         n = (reduceI n       , n + 1)
   15 reduceSkiWithIndex _         _ = error "reduceSkiWithIndex"

   17 reduceApp :: Lambda -> Lambda -> Int -> (Lambda , Int)
   18 reduceApp f a n = (App f' a' , n'') where
   19   (a', n'') = reduceSkiWithIndex a n'
   20   (f', n')  = reduceSkiWithIndex f n

   22 reduceS :: Int -> Lambda
   23 reduceS n = fx $ fy $ fz $ x `App` z `App` App y z where
   24   (fx, x) = makeFunWithVarX n
   25   (fy, y) = makeFunWithVarY n
   26   (fz, z) = makeFunWithVarZ n
   27 
   28 reduceK :: Int -> Lambda
   29 reduceK n = fx $ fy x where
   30   (fx, x) = makeFunWithVarX n
   31   (fy, _) = makeFunWithVarY n
   32 
   33 reduceI :: Int -> Lambda
   34 reduceI n = fx x where
   35   (fx , x) = makeFunWithVarX n

   44 makeFunWithVarZ = makeFunWithVar "z"
