   11 parseLambda :: Text -> Lambda
   12 parseLambda = fst . fst . Unsafe.head . (lambdaParser >*> eof) . lexer . toString
   13 
   14 lambdaParser :: Parse String Lambda
   15 lambdaParser = applyParser `alt` funParser
   16 
   17 applyParser :: Parse String Lambda
   18 applyParser = recL1 App atomParser
   19 
   20 funParser :: Parse String Lambda
   21 funParser = token "\\" >*> paramsParser >*> token "->" >*> lambdaParser `build` buildFun
   22 
   23 buildFun :: (String, ([String], (String, Lambda))) -> Lambda
   24 buildFun ("\\", (ps, ("->", e))) = makeFun (toText <$> ps) e
   25 buildFun  _                      = error "buildFun"
   26 
   27 makeFun :: [Text] -> Lambda -> Lambda
   28 makeFun [p] ex      = Abs p ex
   29 makeFun (p : ps) ex = Abs p $ makeFun ps ex
   30 makeFun _ _         = error "makeFun error: need 1 parameter at least"
   31 
   32 atomParser :: Parse String Lambda
   33 atomParser = (spot (isLower . Unsafe.head) `build` (Var . toText)) `alt` parensParser
   34 
   35 parensParser :: Parse String Lambda
   36 parensParser = token "(" >*> lambdaParser >*> token ")" `build` fst . snd
   37 
   38 paramsParser :: Parse String [String]
   39 paramsParser = list1 $ spot $ isLower . Unsafe.head
   40 
   41 lexer :: String -> [String]
   42 lexer             ""  = []
   43 lexer (' '       : s) = lexer s
   44 lexer ('\n'      : s) = lexer s
   45 lexer ('\\'      : s) = "\\" : lexer s
   46 lexer ('-' : '>' : s) = "->" : lexer s
   47 lexer (      '(' : s) = "("  : lexer s
   48 lexer (      ')' : s) = ")"  : lexer s
   49 lexer ca@(c : _)
   50   | isLower c    = ret : lexer s where (ret, s) = span isAlphaNum ca
   51 lexer ca      = error $ "lexer error: " <> toText ca
