   11 reduceAbstractionsForRoot :: Instruction -> Instruction
   12 reduceAbstractionsForRoot (Def n f) = Def n $ reduceAbstractions f
   13 reduceAbstractionsForRoot (Eval  f) = Eval  $ reduceAbstractions f

   15 reduceAbstractions :: Lambda -> Lambda
   16 reduceAbstractions (App f g) = App (reduceAbstractions f) (reduceAbstractions g)
   17 reduceAbstractions (Abs n f) = reduceToSki n f
   18 reduceAbstractions  l        = l

   22 onlySKI :: Lambda -> Bool
   23 onlySKI S         = True
   24 onlySKI K         = True
   25 onlySKI I         = True
   26 onlySKI (App f a) = onlySKI f && onlySKI a
   27 onlySKI _         = False
   28 
   29 reduceToSki :: Text -> Lambda -> Lambda
   30 reduceToSki _ l
   31   | onlySKI l             = App K l
   32 reduceToSki x0 (App f  a) = App (App S $ reduceToSki x0 f) $ reduceToSki x0 a
   33 reduceToSki x0 (Abs x1 e) = reduceAbs x0 x1 e
   34 reduceToSki x0 (Var x1  ) = reduceVar x0 x1
   35 reduceToSki _   lambda    = lambda
   36 
   37 reduceAbs :: Text -> Text -> Lambda -> Lambda
   38 reduceAbs x0 x1 e
   39   | x0 == x1  = App K $ reduceToSki x0 e
   40   | otherwise = reduceToSki x0 $ reduceToSki x1 e
   41 
   42 reduceVar :: Text -> Text -> Lambda
   43 reduceVar x0 x1
   44   | x1 == x0  = I
   45   | otherwise = App K (Var x1)
